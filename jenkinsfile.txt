pipeline {
    agent any
    tools{
        maven 'M2'
    }
    stages {
        stage('Fetching') {
            steps {
                git branch:'main',url:'https://github.com/Isha-nagpal/ISHA_SpringPetClinic'
            }
        }
        stage('Compile') {
            steps {
                bat 'mvn compile'
            }
        }
        stage('Test') {
            steps {
                bat 'mvn test'
            }
        }
        stage("Package"){
            steps{
                bat 'mvn package'
            }
        }
        stage("Verify"){
            steps{
                bat 'mvn verify'
            }
        }
        stage("Install"){
            steps{
                bat 'mvn install'
            }

    }
}
}
